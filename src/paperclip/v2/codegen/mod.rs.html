29/// Default emitter for anything that implements `Schema` trait.
30///
31/// This doesn't do anything special, as `Emitter` trait methods take
32/// care of all the heavy load.
33pub struct DefaultEmitter<S> {
34    state: EmitterState,
35    _schema: PhantomData<S>,
36}
37
38impl<S> From<EmitterState> for DefaultEmitter<S> {
39    fn from(state: EmitterState) -> Self {
40        DefaultEmitter {
41            state,
42            _schema: PhantomData,
43        }
44    }
45}
46
47impl<S: Schema + Debug> Emitter for DefaultEmitter<S> {
48    type Definition = S;
49
50    fn state(&self) -> &EmitterState {
51        &self.state
52    }
53}

55/// Metadata for generating a crate.
56#[derive(Debug, Default, Clone)]
57#[non_exhaustive]
58pub struct CrateMeta {
59    /// Name of the crate. If this is not specified, then the name of the
60    /// working directory is assumed to be crate name.
61    pub name: Option<String>,
62    /// Version (defaults to 0.1.0)
63    pub version: Option<String>,
64    /// List of authors for this crate. Defaults to cargo's defaults.
65    pub authors: Option<Vec<String>>,
66    /// Whether we're planning to emit a lib, app or module.
67    pub mode: EmitMode,
68    /// Whether or not to make this a root crate.
69    pub no_root: bool,
70}
71
72/// Mode of codegen (module, crate or CLI app).
73#[derive(Debug, Clone, Copy, Eq, PartialEq)]
74pub enum EmitMode {
75    Module,
76    Crate,
77    App,
78}
79
80impl Default for EmitMode {
81    fn default() -> Self {
82        // NOTE: Module by default - CLI will specify other modes if needed.
83        EmitMode::Module
84    }
