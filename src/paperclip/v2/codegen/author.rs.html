3//! Obtained from [Cargo](https://github.com/rust-lang/cargo/blob/fa05862cd0c6b899b801fda0f256ac5b9bae69d9/src/cargo/ops/cargo_new.rs#L690-L750).

5use anyhow::Error;
6use git2::{Config as GitConfig, Repository as GitRepository};
7
8use std::env;
9
10fn get_environment_variable(variables: &[&str]) -> Option<String> {
11    variables.iter().filter_map(|var| env::var(var).ok()).next()
12}
13
14/// Attempts to find the name and email of the author from the current environment.
15pub(super) fn discover() -> Result<(String, Option<String>), Error> {
16    let cwd = env::current_dir()?;
17    let git_config = if let Ok(repo) = GitRepository::discover(cwd) {
18        repo.config()
19            .ok()
20            .or_else(|| GitConfig::open_default().ok())
21    } else {
22        GitConfig::open_default().ok()
23    };
24    let git_config = git_config.as_ref();
25    let name_variables = [
26        "CARGO_NAME",
27        "GIT_AUTHOR_NAME",
28        "GIT_COMMITTER_NAME",
29        "USER",
30        "USERNAME",
31        "NAME",
32    ];
33    let name = get_environment_variable(&name_variables[0..3])
34        .or_else(|| git_config.and_then(|g| g.get_string("user.name").ok()))
35        .or_else(|| get_environment_variable(&name_variables[3..]));
36
37    let name = match name {
38        Some(name) => name,
39        None => {
40            let username_var = if cfg!(windows) { "USERNAME" } else { "USER" };
41            anyhow::bail!(
42                "could not determine the current user, please set ${}",
43                username_var
44            )
45        }
46    };
47    let email_variables = [
48        "CARGO_EMAIL",
49        "GIT_AUTHOR_EMAIL",
50        "GIT_COMMITTER_EMAIL",
51        "EMAIL",
52    ];
53    let email = get_environment_variable(&email_variables[0..3])
54        .or_else(|| git_config.and_then(|g| g.get_string("user.email").ok()))
55        .or_else(|| get_environment_variable(&email_variables[3..]));
56
57    let name = name.trim().to_string();
58    let email = email.map(|s| {
59        let mut s = s.trim();
60
61        // In some cases emails will already have <> remove them since they
62        // are already added when needed.
63        if s.starts_with('<') && s.ends_with('>') {
64            s = &s[1..s.len() - 1];
65        }
66
67        s.to_string()
68    });
69
70    Ok((name, email))
71}
